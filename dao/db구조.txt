CREATE TABLE `gppl_chatmessage` (
	`MESSAGE_NUM` BIGINT(20) NOT NULL DEFAULT '0',
	`ROOM_NUM` INT(11) NOT NULL,
	`USER_ID` VARCHAR(30) NULL DEFAULT NULL COLLATE 'utf8_general_ci',
	`MESSAGE_TEXT` VARCHAR(3000) NULL DEFAULT NULL COLLATE 'utf8_general_ci',
	PRIMARY KEY (`ROOM_NUM`, `MESSAGE_NUM`) USING BTREE
)

CREATE TABLE `gppl_chatroom` (
	`ROOM_NUM` INT(11) NOT NULL,
	`USER_TYPE` CHAR(1) NOT NULL COLLATE 'utf8_general_ci',
	`USER_ID` VARCHAR(30) NULL DEFAULT NULL COLLATE 'utf8_general_ci',
	`VIEW_MESSAGE_NUM` BIGINT(20) NULL DEFAULT NULL,
	`END_MESSAGE_NUM` BIGINT(20) NULL DEFAULT NULL,
	`DELETE_FLAG` CHAR(1) NULL DEFAULT '0' COLLATE 'utf8_general_ci',
	`CREATED_DATE` TIMESTAMP NULL DEFAULT current_timestamp(),
	`UPDATED_DATE` TIMESTAMP NOT NULL DEFAULT '0000-00-00 00:00:00',
	PRIMARY KEY (`ROOM_NUM`, `USER_TYPE`) USING BTREE
)
